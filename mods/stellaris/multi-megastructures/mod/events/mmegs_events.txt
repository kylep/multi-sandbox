namespace = mmegs

# Grants every regular empire (player + AI) every megastructure technology at game start.
# Fires once per country on game start via on_game_start_country.
country_event = {
	id = mmegs.1
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		is_country_type = default
	}

	immediate = {
		# Base + always-loaded megastructure techs (Utopia / Federations /
		# MegaCorp). These resolve for every supported configuration.
		give_technology = { tech = tech_mega_engineering message = no }
		give_technology = { tech = tech_dyson_sphere message = no }
		give_technology = { tech = tech_ring_world message = no }
		give_technology = { tech = tech_matter_decompressor message = no }
		give_technology = { tech = tech_science_nexus message = no }
		give_technology = { tech = tech_sentry_array message = no }
		give_technology = { tech = tech_strategic_coordination message = no }
		give_technology = { tech = tech_mega_art message = no }
		give_technology = { tech = tech_interstellar_assembly message = no }
		give_technology = { tech = tech_mega_shipyard message = no }
		give_technology = { tech = tech_quantum_catapult message = no }

		# DLC-gated megastructure techs. Each grant is wrapped in its DLC check
		# because give_technology on a tech whose definition didn't load (DLC
		# absent) throws "invalid technology" into error.log for every empire,
		# every game start. Guarding keeps the log clean for players who don't
		# own the pack.
		if = {
			limit = { has_machine_age_dlc = yes }
			give_technology = { tech = tech_dyson_swarm message = no }
			give_technology = { tech = tech_orbital_arc_furnace message = no }
		}
		if = {
			limit = { has_grand_archive_dlc = yes }
			give_technology = { tech = tech_galactic_archivism message = no }
		}
		if = {
			limit = { has_biogenesis_dlc = yes }
			give_technology = { tech = tech_deep_space_citadel message = no }
		}
		if = {
			limit = { has_nomads_dlc = yes }
			give_technology = { tech = tech_dyson_gun message = no }
		}

		# tech_cosmogenesis_world is intentionally NOT granted. Its potential is
		# `has_machine_age_dlc = yes AND has_ascension_perk = ap_cosmogenesis`, so
		# give_technology rejects it at game start regardless of DLC — it can only
		# be obtained on the Cosmogenesis crisis-ascension path, not here.
	}
}
